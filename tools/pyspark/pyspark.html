Dağıtık Analitik Motoru

Python ortamında bir SparkSession oluşturarak milyarlarca satırlık CSV dosyalarını saniyeler içerisinde sorgulayabilirsiniz. 

from pyspark.sql import SparkSession
from pyspark.sql.functions import col, desc

# Spark Session Başlatma
spark = SparkSession.builder \
    .appName("BigDataAnalytics") \
    .getOrCreate()

# Dağıtık HDFS'ten Devasa CSV Okuma
df = spark.read.csv("hdfs://cluster/data/transactions.csv", header=True, inferSchema=True)

# Veri Filtreleme ve Gruplama (Dağıtık İşlem)
top_customers = df.filter(col("status") == "COMPLETED") \
                  .groupBy("customer_id") \
                  .sum("amount") \
                  .orderBy(desc("sum(amount)"))

# Sonuçları Görüntüleme (Action)
top_customers.show(5)

spark.stop()
